def calcular_indicadores(df_spark):
    """
    Calcula los indicadores principales para Caf√© Andino
    """
    df = df_spark.toPandas()
    fecha_actual = datetime.now().strftime("%d/%m/%Y %H:%M")
    
    # M√©tricas principales
    total_registros = len(df)
    total_ventas_unicas = df['venta_id'].nunique() if 'venta_id' in df.columns else total_registros
    total_clientes = df['cliente_id'].nunique() if 'cliente_id' in df.columns else 0
    total_colaboradores = df['colaborador_id'].nunique() if 'colaborador_id' in df.columns else 0
    total_brokers = df['broker_aduanero_id'].nunique() if 'broker_aduanero_id' in df.columns else 0
    
    # An√°lisis de fechas
    if 'fecha_de_venta' in df.columns:
        df['fecha_de_venta'] = pd.to_datetime(df['fecha_de_venta'])
        fecha_min = df['fecha_de_venta'].min().strftime("%d/%m/%Y")
        fecha_max = df['fecha_de_venta'].max().strftime("%d/%m/%Y")
        
        # Ventas √∫ltimos 30 d√≠as
        fecha_limite = datetime.now() - timedelta(days=30)
        ventas_recientes = len(df[df['fecha_de_venta'] >= fecha_limite])
    else:
        fecha_min = "N/A"
        fecha_max = "N/A"
        ventas_recientes = 0
    
    metricas = {
        'total_registros': total_registros,
        'total_ventas': total_ventas_unicas,
        'total_clientes': total_clientes,
        'total_colaboradores': total_colaboradores,
        'total_brokers': total_brokers,
        'ventas_ultimos_30_dias': ventas_recientes,
        'fecha_reporte': fecha_actual,
        'fecha_min': fecha_min,
        'fecha_max': fecha_max
    }
    
    return metricas, df

def enviar_correo_gmail(subject, html_content):
    """
    Env√≠a el correo electr√≥nico con el reporte usando Gmail
    """
    try:
        print("üìß Configurando env√≠o por Gmail...")
        msg = MIMEMultipart('alternative')
        msg['From'] = EMAIL_SENDER
        msg['To'] = EMAIL_RECIPIENT
        msg['Subject'] = subject
        
        html_part = MIMEText(html_content, 'html')
        msg.attach(html_part)
        
        print("üîó Conectando con servidor Gmail...")
        server = smtplib.SMTP(SMTP_SERVER, SMTP_PORT)
        server.ehlo()
        server.starttls()
        server.ehlo()
        
        print("üîê Iniciando sesi√≥n con App Password...")
        server.login(EMAIL_SENDER, EMAIL_PASSWORD)
        
        print("üì§ Enviando correo...")
        server.send_message(msg)
        server.quit()
        
        print("‚úÖ Correo enviado exitosamente por Gmail")
        return True
    except Exception as e:
        print(f"‚ùå Error al enviar correo por Gmail: {str(e)}")
        return False
